#include<stdio.h>
#include<iostream>
#include<queue>
using namespace std;
int arr[1001][1001];
int visit[1001][1001];
int mindis = 0;
int n, m;
void dfs(int now,int des,int dis) {
    if (now == des)
    {
        if (mindis > dis)
        {
            mindis = dis;
        }
        return;
    }
    for (int k = 1; k <= n; k++)
    {
        if (arr[now][k] != 0 && visit[now][k]!=1)
        {
            visit[now][k] = 1;
            visit[k][now] = 1;
            dfs(k, des, dis + arr[now][k]);
        }
    }
}
int main() {
    cin >> n >> m;
    for (int k = 0; k < n - 1; k++)
    {
        int a, b,l;
        cin >> a >> b>>l;
        arr[a][b] = l;
        arr[b][a] = l;
    }
    for (int k = 0; k < m; k++)
    {
        int a, b;
        cin >> a >> b;
        if (arr[a][b] != 0)
        {
            cout << arr[a][b] << "\n";
            continue;
        }
        mindis = 9999999;
        for (int k = 1; k <= n; k++)
        {
            for (int i = 1; i <= n; i++)
            {
                visit[k][i] = 0;
            }
        }
        dfs(a, b,0);
        cout << mindis << "\n";
    }
}

깊이 우선 탐색(DFS, Depth-First Search)

깊이 우선 탐색의 개념

너비 우선 탐색과 다르게 다음 분기로 넘어가기 전에 해당 분기를 완벽하게 탐색하는 방법


깊이 우선 탐색의 특징

  • 모든 노드를 방문 해야 할 때 사용한다.
  • 구현 방식은 BFS에 비해서 간단하다.
  • 재귀의 구조로 구현된다.(자기 자신을 계속 호출)
  • 방문 체크를 하지 않으면 무한 루프에 빠질 수 있다.


깊이 우선 탐색의 시간 복잡도

  • 인접 리스트로 구현한 경우 : O(N+E) (E는 간선의 개수)
  • 인접 행렬로 구현한 경우 : O(N^2)
