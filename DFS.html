#include<stdio.h>
#include<iostream>
#include<queue>
using namespace std;
int arr[1001][1001];
int visit[1001][1001];
int mindis = 0;
int n, m;
void dfs(int now,int des,int dis) {
    if (now == des)
    {
        if (mindis > dis)
        {
            mindis = dis;
        }
        return;
    }
    for (int k = 1; k <= n; k++)
    {
        if (arr[now][k] != 0 && visit[now][k]!=1)
        {
            visit[now][k] = 1;
            visit[k][now] = 1;
            dfs(k, des, dis + arr[now][k]);
        }
    }
}
int main() {
    cin >> n >> m;
    for (int k = 0; k < n - 1; k++)
    {
        int a, b,l;
        cin >> a >> b>>l;
        arr[a][b] = l;
        arr[b][a] = l;
    }
    for (int k = 0; k < m; k++)
    {
        int a, b;
        cin >> a >> b;
        if (arr[a][b] != 0)
        {
            cout << arr[a][b] << "\n";
            continue;
        }
        mindis = 9999999;
        for (int k = 1; k <= n; k++)
        {
            for (int i = 1; i <= n; i++)
            {
                visit[k][i] = 0;
            }
        }
        dfs(a, b,0);
        cout << mindis << "\n";
    }
}
